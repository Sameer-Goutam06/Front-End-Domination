Units:
vw- view port width
vh- view port height
vmin- minimum value in viewport(laptop-height,mobile-width)
vmax-maximum value in viewport
px-pixels
%-percent from parent

flex-properties:
flex-direction:row or column
flex-wrap:wrap or no-wrap // doesnt let the flex property to reduce the space alloted to items
flex-shrink:0 or above //does the same thing as flex-wrap but flex-wrap is applied to parent and this is applied to the item itself
justify-content:space-between,space-around,flex-start,flex-end,center
align-items:center,flex-start,flex-end,baseline,stretch
